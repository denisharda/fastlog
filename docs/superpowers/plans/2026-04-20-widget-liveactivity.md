# Widget & Live Activity Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Fix the blank FastLog home-screen widget and silent Live Activity by pushing state through `expo-widgets`' first-class APIs, and redesign both surfaces in the Amber Sunrise "Phase Ring Forward" / "Ring Echo" directions approved in the spec.

**Architecture:** JS calls `Widget.updateTimeline(...)` on fast start and `Widget.updateSnapshot(...)` on every state change. Live Activities are started via the factory exported directly from `widgets/FastingActivity.tsx` (no stub duplicate). The App Groups UserDefaults bridge is removed entirely — `expo-widgets` props are the single source of truth.

**Tech Stack:** React Native (Expo SDK 55), TypeScript strict, `expo-widgets` 55.0.7, `@expo/ui/swift-ui` (Gauge, Circle, ZStack), Zustand, React Query.

**Spec:** `docs/superpowers/specs/2026-04-20-widget-liveactivity-design.md`

**Context for the engineer:**
- `expo-widgets` compiles the default-exported `Widget<T>` into a native iOS widget extension at prebuild time. The component function body must be marked with the `'widget'` directive (string literal on the first line of the function body). Only primitives imported from `@expo/ui/swift-ui` + its `modifiers` subpath are renderable — no regular React Native views.
- Colour constants must be **local to this file** because the `'widget'` boundary can't import hooks or dynamic modules. Keep `WIDGET_COLORS` in sync with `constants/theme.ts` — a comment at the top of the file flags this.
- `Text date={...} dateStyle="timer"` is a SwiftUI feature that auto-updates every minute without JS involvement. Keep using it.
- `Gauge` + `gaugeStyle('circularCapacity')` + `tint(color)` renders a solid-colour progress ring; put the timer text inside via `currentValueLabel`. The ring tint shifts per phase — this is intentional and matches the Amber Sunrise phase spectrum.

## Task 3: Create the widget JS helper

**Files:**
- Create: `lib/widget.ts`

**Context:**
- `expo-widgets` exposes `Widget<T>.updateSnapshot(props)` (immediate one-off render) and `Widget<T>.updateTimeline(entries)` (pre-schedules N future render points). Both are fire-and-forget — no Promise, no error propagation. We wrap both in `try/catch` and an iOS-only guard.
- The widget factory is imported directly from `widgets/FastingWidget.tsx` — same JS module can be imported from both widget extension and main app bundle, that's how `expo-widgets` is designed.
- We **don't** need `Platform.OS !== 'ios'` reruns on every call because the whole module is a no-op on Android; but we still guard for Expo Go (no native module available).

- [ ] **Step 3.1: Create `lib/widget.ts`**

```ts
import { Platform } from 'react-native';
import Constants, { ExecutionEnvironment } from 'expo-constants';
import FastingWidget, { type FastingState } from '../widgets/FastingWidget';

const isExpoGo = Constants.executionEnvironment === ExecutionEnvironment.StoreClient;

export type { FastingState } from '../widgets/FastingWidget';

function available(): boolean {
  return Platform.OS === 'ios' && !isExpoGo;
}

/**
 * Immediately push a single render of the widget. Use on state changes
 * (phase transitions, app foreground, fast start/stop).
 */
export function pushWidgetSnapshot(state: FastingState): void {
  if (!available()) return;
  try {
    FastingWidget.updateSnapshot(state);
  } catch (e) {
    console.warn('[widget] updateSnapshot failed:', e);
  }
}

/**
 * Pre-schedule hourly timeline entries for an active fast. Ring fill /
 * percent stays fresh without needing JS to wake up.
 *
 * Generates entries at startedAt + 0h, 1h, 2h, ..., targetHours.
 * The final entry is at targetHours with the "Deep Fast" phase.
 */
export function scheduleWidgetTimeline(args: {
  startedAt: string;
  targetHours: number;
  protocol: string;
}): void {
  if (!available()) return;
  const { startedAt, targetHours, protocol } = args;
  try {
    const start = new Date(startedAt);
    const entries = [];
    for (let h = 0; h <= Math.max(1, Math.ceil(targetHours)); h++) {
      const date = new Date(start.getTime() + h * 3600_000);
      entries.push({
        date,
        props: {
          isActive: true,
          startedAt,
          targetHours,
          phase: phaseNameForHour(h),
          protocol,
        } satisfies FastingState,
      });
    }
    FastingWidget.updateTimeline(entries);
  } catch (e) {
    console.warn('[widget] updateTimeline failed:', e);
  }
}

/**
 * Clear the widget back to the inactive layout.
 */
export function clearWidgetSnapshot(lastProtocol: string): void {
  if (!available()) return;
  try {
    FastingWidget.updateSnapshot({
      isActive: false,
      startedAt: null,
      targetHours: 0,
      phase: 'Fed State',
      protocol: lastProtocol,
    });
  } catch (e) {
    console.warn('[widget] clearWidgetSnapshot failed:', e);
  }
}

const PHASE_BY_MIN_HOUR: readonly [number, string][] = [
  [18, 'Deep Fast'],
  [16, 'Autophagy Zone'],
  [12, 'Fat Burning Peak'],
  [8, 'Fat Burning Begins'],
  [4, 'Early Fasting'],
  [0, 'Fed State'],
];

function phaseNameForHour(h: number): string {
  for (const [min, name] of PHASE_BY_MIN_HOUR) {
    if (h >= min) return name;
  }
  return 'Fed State';
}
```

## Task 4: Rewrite `lib/liveActivity.ts` to use the real factory

**Files:**
- Modify: `lib/liveActivity.ts` (full rewrite)

**Context:**
- The previous file called `mod.createLiveActivity('FastingActivity', stubComponent)` — this creates a **second factory instance** with the same name, which conflicts with the real factory exported from `widgets/FastingActivity.tsx`. The native side has only one registered activity type, so the stub factory's `start()` may succeed but render nothing (because the real activity attributes never flow through), or crash silently on type mismatch.
- Factory instances persist across app state changes; `getInstances()` returns an array of live activity handles that iOS is currently displaying. We use this to reattach on cold start.
- `start()` signature per the `expo-widgets` types: `start(props: T, url?: string): LiveActivity<T>` — **synchronous**.

- [ ] **Step 4.1: Rewrite `lib/liveActivity.ts`**

Full replacement contents:

```ts
import { Platform } from 'react-native';
import Constants, { ExecutionEnvironment } from 'expo-constants';
import type { LiveActivity as LiveActivityInstance } from 'expo-widgets';
import FastingActivityFactory, { type FastingActivityState } from '../widgets/FastingActivity';

const isExpoGo = Constants.executionEnvironment === ExecutionEnvironment.StoreClient;

export type { FastingActivityState } from '../widgets/FastingActivity';
// Legacy alias — some callers reference the old name.
export type LiveActivityState = FastingActivityState;

let activity: LiveActivityInstance<FastingActivityState> | null = null;

function available(): boolean {
  return Platform.OS === 'ios' && !isExpoGo;
}

/**
 * Start a new Live Activity. Ends any existing instance first.
 */
export async function startLiveActivity(state: FastingActivityState): Promise<void> {
  if (!available()) return;

  await endLiveActivity();

  try {
    activity = FastingActivityFactory.start(state, 'fastlog://timer');
    if (__DEV__) console.log('[liveActivity] started');
  } catch (e) {
    console.error('[liveActivity] start failed:', e);
    activity = null;
  }
}

/**
 * Update the current Live Activity's content. No-op if none running.
 */
export async function updateLiveActivity(state: FastingActivityState): Promise<void> {
  if (!activity) return;
  try {
    await activity.update(state);
  } catch (e) {
    console.warn('[liveActivity] update failed:', e);
  }
}

/**
 * End the current Live Activity with the default dismissal policy.
 */
export async function endLiveActivity(): Promise<void> {
  if (!activity) return;
  try {
    await activity.end('default');
  } catch (e) {
    console.warn('[liveActivity] end failed:', e);
  } finally {
    activity = null;
  }
}

/**
 * Reattach to an existing Live Activity on app cold start, or start a
 * new one if none are running.
 */
export async function restoreLiveActivity(state: FastingActivityState): Promise<void> {
  if (!available()) return;

  try {
    const existing = FastingActivityFactory.getInstances();
    if (existing.length > 0) {
      activity = existing[0];
      await updateLiveActivity(state);
      if (__DEV__) console.log('[liveActivity] reattached');
      return;
    }
  } catch (e) {
    console.warn('[liveActivity] getInstances failed:', e);
  }

  await startLiveActivity(state);
}

export function hasLiveActivity(): boolean {
  return activity !== null;
}
```
